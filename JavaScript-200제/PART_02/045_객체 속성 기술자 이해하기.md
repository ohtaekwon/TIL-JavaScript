###  :pencil: ***입문 045 :  객체 속성 기술자 이해하기***

<br>

```javascript
let user = {
    name : 'TaeKwon'
};
let descriptor = Object.getOwnPropertyDescriptor(user, 'name');
console.log(descriptor);		// {value: 'TAEKWON', writable: true, enumerable: true, configurable: true}

let user2 = {};
Object.defineProperty(user2, "name", {
    value : 'TaeKwon2',
    enumerable : true,
    configurable : true,
    writable : false,
});								// {name: 'TaeKwon'}
console.log(user2.name);		// TaeKwon
user2.name="OH";				// 'OH'
console.log(user2.name);		// TaeKwon

let user3 = {
    name : "TaeKwon3",
    toString(){
        return this.name;
    }
};
Object.defineProperty(user3, "toString", {
    enumerable : false
});								
// {name: 'TaeKwon3', toString: ƒ}name: "TaeKwon3"toString: ƒ toString()[[Prototype]]: Object

for (let key in user3) {
    console.log(key);
}								// name

let user4 = {};
Object.defineProperty(user4, "name", {
    value : "TaeKwon4",
    configurable : false,
});

delete user4.name
console.log(user4);
Object.defineProperty(user4, "name", {
    writable : true,
});
```
